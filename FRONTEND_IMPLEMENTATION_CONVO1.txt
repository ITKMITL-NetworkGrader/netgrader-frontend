const remaining = computed(() => {
  const targetTime = targetDate.value.getTime();
  const now = currentTime.value;
  
  setTimerInterval();

  if (targetTime <= 0) return "Unlimited";
  const diff = targetTime - now;
  if (diff <= 0) return "Time's up!";

  const hours = Math.floor(diff / (1000 * 60 * 60));
  const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
  const seconds = Math.floor((diff % (1000 * 60)) / 1000);
  const milliseconds = diff % 1000;

  if (import.meta.dev) {
    return `${hours.toString().padStart(2, "0")}:${minutes
      .toString()
      .padStart(2, "0")}:${seconds.toString().padStart(2, "0")}.${milliseconds
      .toString()
      .padStart(3, "0")}`;
  }

  return `${hours.toString().padStart(2, "0")}:${minutes
    .toString()
    .padStart(2, "0")}:${seconds.toString().padStart(2, "0")}`;
});

const isTimerOpen = ref(true);

const setTimerInterval = () => {
  timerInt.value = setInterval(() => {
    currentTime.value = Date.now();
    if (targetDate.value.getTime() <= currentTime.value) {
      clearInterval(timerInt.value);
    }
  }, import.meta.dev ? 25 : 500); // Reduced interval for smoother milliseconds display
}

onMounted(() => {
  setTimerInterval();

  timerUpdater.value = setInterval(() => {
    props.timerRefresh();
  }, 5000);
});

onBeforeUnmount(() => {
  clearInterval(timerInt.value);
  clearInterval(timerUpdater.value);
});
